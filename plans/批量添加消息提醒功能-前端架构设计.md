## 1. 功能概述

### 1.1 功能描述

用户可以一次选择多个股票或基金，然后为它们批量设置相同类型的价格提醒，包括：
- **价格上限提醒**：当价格超过设定阈值时触发
- **价格下限提醒**：当价格低于设定阈值时触发
- **涨跌幅提醒**：当涨跌幅达到设定百分比时触发

### 1.2 核心用户流程

```
选择资产类型（股票/基金） 
  → 批量选择标的（支持搜索、筛选、多选）
  → 选择提醒类型（价格上限/下限/涨跌幅）
  → 填写提醒参数
  → 设置通知方式（站内消息/邮件/推送）
  → 预览确认
  → 提交创建
  → 查看创建结果
```

## 3. 状态管理设计 (Pinia)

### 3.1 Store 结构

```typescript
// stores/batchAlert.ts
import { defineStore } from 'pinia'
import type { 
  AlertType, 
  NotificationMethod, 
  SelectedAsset,
  AlertConfig,
  BatchCreateResult 
} from '@/types/alert'

interface BatchAlertState {
  // 步骤控制
  currentStep: number
  steps: Array<{ key: string; title: string; completed: boolean }>
  
  // 资产选择
  assetType: 'stock' | 'fund'
  selectedAssets: SelectedAsset[]
  
  // 提醒配置
  alertType: AlertType | null
  alertConfig: AlertConfig
  notificationMethods: NotificationMethod[]
  
  // 创建结果
  createResult: BatchCreateResult | null
  
  // UI状态
  isSubmitting: boolean
  error: string | null
}

export const useBatchAlertStore = defineStore('batchAlert', {
  state: (): BatchAlertState => ({
    currentStep: 0,
    steps: [
      { key: 'asset', title: '选择资产', completed: false },
      { key: 'config', title: '配置提醒', completed: false },
      { key: 'preview', title: '预览确认', completed: false },
      { key: 'result', title: '创建结果', completed: false },
    ],
    assetType: 'stock',
    selectedAssets: [],
    alertType: null,
    alertConfig: {
      priceUpper: null,
      priceLower: null,
      changePercent: null,
    },
    notificationMethods: ['in_app'],
    createResult: null,
    isSubmitting: false,
    error: null,
  }),

  getters: {
    // 当前步骤配置
    currentStepConfig: (state) => state.steps[state.currentStep],
    
    // 是否可以进入下一步
    canProceed: (state): boolean => {
      switch (state.currentStep) {
        case 0: return state.selectedAssets.length > 0
        case 1: return state.alertType !== null && validateAlertConfig(state)
        case 2: return true
        default: return false
      }
    },
    
    // 是否可以提交
    canSubmit: (state): boolean => {
      return state.selectedAssets.length > 0 
        && state.alertType !== null 
        && validateAlertConfig(state)
    },
    
    // 已选资产按类型分组
    selectedAssetsByType: (state) => {
      return state.selectedAssets.reduce((acc, asset) => {
        const type = asset.type
        if (!acc[type]) acc[type] = []
        acc[type].push(asset)
        return acc
      }, {} as Record<string, SelectedAsset[]>)
    },
  },

  actions: {
    // 步骤导航
    nextStep() {
      if (this.canProceed && this.currentStep < this.steps.length - 1) {
        this.steps[this.currentStep].completed = true
        this.currentStep++
      }
    },
    
    prevStep() {
      if (this.currentStep > 0) {
        this.currentStep--
      }
    },
    
    goToStep(step: number) {
      if (step >= 0 && step < this.steps.length) {
        // 只能跳转到已完成的步骤或当前步骤的下一步
        if (step <= this.currentStep + 1) {
          this.currentStep = step
        }
      }
    },
    
    // 资产操作
    addAssets(assets: SelectedAsset[]) {
      const existingCodes = new Set(this.selectedAssets.map(a => a.code))
      const newAssets = assets.filter(a => !existingCodes.has(a.code))
      this.selectedAssets.push(...newAssets)
    },
    
    removeAsset(code: string) {
      const index = this.selectedAssets.findIndex(a => a.code === code)
      if (index > -1) {
        this.selectedAssets.splice(index, 1)
      }
    },
    
    clearAssets() {
      this.selectedAssets = []
    },
    
    // 提醒配置
    setAlertType(type: AlertType) {
      this.alertType = type
      // 切换类型时重置配置
      this.alertConfig = {
        priceUpper: null,
        priceLower: null,
        changePercent: null,
      }
    },
    
    updateAlertConfig(config: Partial<AlertConfig>) {
      this.alertConfig = { ...this.alertConfig, ...config }
    },
    
    setNotificationMethods(methods: NotificationMethod[]) {
      this.notificationMethods = methods
    },
    
    // 提交创建
    async submitBatchCreate(): Promise<BatchCreateResult> {
      this.isSubmitting = true
      this.error = null
      
      try {
        const result = await createBatchAlerts({
          assets: this.selectedAssets,
          alertType: this.alertType!,
          config: this.alertConfig,
          notificationMethods: this.notificationMethods,
        })
        
        this.createResult = result
        this.steps[this.currentStep].completed = true
        this.nextStep()
        
        return result
      } catch (error) {
        this.error = error instanceof Error ? error.message : '创建失败'
        throw error
      } finally {
        this.isSubmitting = false
      }
    },
    
    // 重置
    reset() {
      this.currentStep = 0
      this.steps.forEach(s => s.completed = false)
      this.selectedAssets = []
      this.alertType = null
      this.alertConfig = { priceUpper: null, priceLower: null, changePercent: null }
      this.notificationMethods = ['in_app']
      this.createResult = null
      this.error = null
    },
  },
})
```

    ▼

    │                                    │
    │ canProceed?                        │ removeAsset()

[Step 2: 配置提醒] ◄─────────────────────┘

### 3.3 状态持久化策略

对于本功能，**不需要**持久化状态到 localStorage，因为：
- 批量创建是单次操作流程
- 表单数据不适合跨会话保留
- 避免状态污染

如果后续需要"草稿保存"功能，可以扩展：

```typescript
// 草稿持久化（可选扩展）
import { createPersistedStatePlugin } from 'pinia-plugin-persistedstate'

// 在 store 配置中添加
persist: {
  key: 'batch-alert-draft',
  storage: localStorage,
  paths: ['selectedAssets', 'alertConfig', 'notificationMethods'],
}
```

### 4.2 提醒相关 API

```typescript
// services/alert.ts
import http from './http'
import type {
  AlertType,
  NotificationMethod,
  SelectedAsset,
  AlertConfig,
  BatchCreateRequest,
  BatchCreateResult,
  AssetSearchParams,
  AssetSearchResult,
} from '@/types/alert'

export const alertApi = {
  /**
   * 搜索资产（股票/基金）
   */
  searchAssets(params: AssetSearchParams): Promise<AssetSearchResult> {
    return http.get('/assets/search', { params })
  },

  /**
   * 批量创建提醒
   */
  batchCreateAlerts(data: BatchCreateRequest): Promise<BatchCreateResult> {
    return http.post('/alerts/batch-create', data)
  },

  /**
   * 获取资产当前价格（用于表单默认值建议）
   */
  getCurrentPrice(codes: string[]): Promise<Record<string, number>> {
    return http.get('/assets/current-price', { params: { codes: codes.join(',') } })
  },

  /**
   * 验证提醒配置是否有效
   */
  validateAlertConfig(data: {
    assetCode: string
    alertType: AlertType
    config: AlertConfig
  }): Promise<{ valid: boolean; message?: string }> {
    return http.post('/alerts/validate', data)
  },
}

/**
 * 业务层封装（组合 API 调用 + 状态处理）
 */
export async function createBatchAlerts(params: {
  assets: SelectedAsset[]
  alertType: AlertType
  config: AlertConfig
  notificationMethods: NotificationMethod[]
}): Promise<BatchCreateResult> {
  const request: BatchCreateRequest = {
    assetCodes: params.assets.map(a => a.code),
    assetType: params.assets[0].type,
    alertType: params.alertType,
    config: params.config,
    notificationMethods: params.notificationMethods,
  }

  return alertApi.batchCreateAlerts(request)
}
```

// composables/useAssetSearch.ts
import { ref, type Ref } from 'vue'
import { alertApi } from '@/services/alert'
import type { AssetSearchParams, AssetSearchResult, SelectedAsset } from '@/types/alert'

export function useAssetSearch() {
  const loading = ref(false)
  const searchResult = ref<AssetSearchResult | null>(null)

  async function search(params: AssetSearchParams) {
    loading.value = true
    try {
      searchResult.value = await alertApi.searchAssets(params)
    } finally {
      loading.value = false
    }
  }

  function getSelectedAssets(): SelectedAsset[] {
    if (!searchResult.value) return []
    return searchResult.value.items.filter(item => item.selected)
  }

  return {
    loading,
    searchResult,
    search,
    getSelectedAssets,
  }
}

## 5. 表单验证策略

### 5.1 验证规则定义

```typescript
// utils/alertValidator.ts
import type { AlertType, AlertConfig, SelectedAsset } from '@/types/alert'

export interface ValidationResult {
  valid: boolean
  errors: Record<string, string>
}

/**
 * 验证提醒参数
 */
export function validateAlertConfig(
  alertType: AlertType,
  config: AlertConfig
): ValidationResult {
  const errors: Record<string, string> = {}

  switch (alertType) {
    case 'price_upper':
      if (config.priceUpper === null || config.priceUpper === undefined) {
        errors.priceUpper = '请输入价格上限'
      } else if (config.priceUpper <= 0) {
        errors.priceUpper = '价格上限必须大于0'
      }
      break

    case 'price_lower':
      if (config.priceLower === null || config.priceLower === undefined) {
        errors.priceLower = '请输入价格下限'
      } else if (config.priceLower <= 0) {
        errors.priceLower = '价格下限必须大于0'
      }
      break

    case 'change_percent':
      if (config.changePercent === null || config.changePercent === undefined) {
        errors.changePercent = '请输入涨跌幅百分比'
      } else if (Math.abs(config.changePercent) > 100) {
        errors.changePercent = '涨跌幅百分比不能超过100%'
      } else if (config.changePercent === 0) {
        errors.changePercent = '涨跌幅百分比不能为0'
      }
      break
  }

  return {
    valid: Object.keys(errors).length === 0,
    errors,
  }
}

/**
 * 验证已选资产
 */
export function validateSelectedAssets(assets: SelectedAsset[]): ValidationResult {
  const errors: Record<string, string> = {}

  if (assets.length === 0) {
    errors.assets = '请至少选择一个资产'
  } else if (assets.length > 50) {
    errors.assets = '一次最多只能选择50个资产'
  }

  // 验证资产类型一致性
  const types = new Set(assets.map(a => a.type))
  if (types.size > 1) {
    errors.assets = '不能混合选择股票和基金'
  }

  return {
    valid: Object.keys(errors).length === 0,
    errors,
  }
}

/**
 * 智能建议：基于当前价格推荐合理的提醒值
 */
export function suggestAlertValues(
  currentPrice: number,
  alertType: AlertType
): Partial<AlertConfig> {
  switch (alertType) {
    case 'price_upper':
      return { priceUpper: Math.round(currentPrice * 1.05 * 100) / 100 } // 当前价 +5%
    case 'price_lower':
      return { priceLower: Math.round(currentPrice * 0.95 * 100) / 100 } // 当前价 -5%
    case 'change_percent':
      return { changePercent: 5 } // 默认5%
    default:
      return {}
  }
}
```

      label="涨跌幅%"
      prop="changePercent"

      <el-slider
        v-model="formData.changePercent"

        :step="0.1"

    </el-form-item>

    <!-- 通知方式 -->
    <el-form-item label="通知方式">
      <el-checkbox-group v-model="notificationMethods">
        <el-checkbox label="in_app">站内消息</el-checkbox>
        <el-checkbox label="email">邮件</el-checkbox>
        <el-checkbox label="push">推送通知</el-checkbox>
      </el-checkbox-group>
    </el-form-item>
  </el-form>

<script setup lang="ts">
import { ref, computed, watch } from 'vue'
import type { FormInstance, FormRules } from 'element-plus'
import type { AlertType, AlertConfig } from '@/types/alert'
import { suggestAlertValues } from '@/utils/alertValidator'

const props = defineProps<{
  alertType: AlertType
  currentPrice?: number
}>()

const emit = defineEmits<{
  (e: 'update', config: AlertConfig): void
}>()

const formRef = ref<FormInstance>()

const formData = ref<AlertConfig>({
  priceUpper: null,
  priceLower: null,
  changePercent: null,
})

const notificationMethods = ref(['in_app'])

// 表单验证规则
const formRules = computed<FormRules>(() => ({
  priceUpper: [
    { required: props.alertType === 'price_upper', message: '请输入价格上限', trigger: 'blur' },
    { type: 'number', min: 0.01, message: '价格必须大于0', trigger: 'blur' },
  ],
  priceLower: [
    { required: props.alertType === 'price_lower', message: '请输入价格下限', trigger: 'blur' },
    { type: 'number', min: 0.01, message: '价格必须大于0', trigger: 'blur' },
  ],
  changePercent: [
    { required: props.alertType === 'change_percent', message: '请输入涨跌幅百分比', trigger: 'change' },
    { 
      validator: (_rule, value, callback) => {
        if (Math.abs(value) > 100) {
          callback(new Error('不能超过100%'))
        } else if (value === 0) {
          callback(new Error('不能为0'))
        } else {
          callback()
        }
      }, 
      trigger: 'change' 
    },
  ],
}))

// 智能建议值
const suggestedValues = computed(() => {
  if (!props.currentPrice) return {}
  return suggestAlertValues(props.currentPrice, props.alertType)
})

// 监听表单数据变化，同步到 store
watch(
  [formData, notificationMethods],
  () => {
    emit('update', {
      ...formData.value,
      notificationMethods: notificationMethods.value,
    })
  },
  { deep: true }
)

// 暴露验证方法
defineExpose({
  async validate() {
    return formRef.value?.validate()
  },
})
</script>

    }

// composables/useDebounceSearch.ts
import { ref, watch } from 'vue'
import { debounce } from 'lodash-es'

export function useDebounceSearch(searchFn: (keyword: string) => void, delay = 300) {
  const keyword = ref('')
  
  const debouncedSearch = debounce((value: string) => {
    searchFn(value)
  }, delay)

  watch(keyword, (newValue) => {
    debouncedSearch(newValue)
  })

  return { keyword }
}
